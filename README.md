# Requires (listed in requirements.tx)
- behave
- nose

# Project setup
    $ git clone https://github.com/AndreasAugustin/Gherkin-Demos-python.git
    $ cd Gherkin-Demos-python
    $ pip install -r requirements.txt

# Run tests
    $ behave ./test/features/
